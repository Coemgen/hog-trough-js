/**
 * Current date in UTC format.
 * @returns {object} The current UTC date object.
 */
let utcDate = (function () {
    "use strict";
    let d = new Date();
    let yyyy = d.getFullYear();
    let mm = d.getMonth() + 1;
    let dd = d.getDate();
    let hh = d.getHours();
    let MM = d.getMinutes();
    let utcDate = yyyy + "-" +
        mm.toString().padStart(2, "0") + "-" +
        dd.toString().padStart(2, "0") +
        "T" +
        hh.toString().padStart(2, "0") + ":" +
        MM.toString().padStart(2, "0") + ":" +
        "00";
    return {
        get: function () {
            return utcDate;
        }
    };
}());
